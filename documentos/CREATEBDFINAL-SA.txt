CREATE TABLE sistemadeavaliacao.FUNCIONARIO (
  ID_FUNCIONARIO SERIAL NOT NULL,
  NOME VARCHAR(50) NOT NULL,
  EMAIL VARCHAR(50) NOT NULL,
  QTD_PROVAS_CONC INTEGER NULL,
  QTD_PROVAS_PEN INTEGER NULL,
  SENHA VARCHAR(20) NOT NULL,
  ADMIN BOOL NULL,
  AVALIADOR BOOL NULL,
  ESPECIALIDADE INTEGER NULL,
  PRIMARY KEY(ID_FUNCIONARIO)
);

CREATE TABLE sistemadeavaliacao.LINGUAGEM (
  ID_LINGUAGEM SERIAL NOT NULL,
  DESCRICAO VARCHAR NOT NULL,
  PRIMARY KEY(ID_LINGUAGEM)
);

CREATE TABLE sistemadeavaliacao.PROVA (
  ID_PROVA SERIAL NOT NULL,
  CANDIDATO_ID INTEGER NOT NULL,
  LINGUAGEM_ID INTEGER NOT NULL,
  LINK_PROVA VARCHAR NOT NULL,
  PRAZO VARCHAR(20) NOT NULL,
  DATA_SOLIC_AVA DATE NOT NULL,
  PRIMARY KEY(ID_PROVA)
);

CREATE TABLE sistemadeavaliacao.CRITERIOS_RESPONDIDOS (
  ID_CRITERIOS_RESPONDIDOS SERIAL NOT NULL,
  CRITERIO_ID INTEGER NOT NULL,
  LINGUAGEM_ID INTEGER NOT NULL,
  AVALIACAO_ID INTEGER NOT NULL,
  RESPONDIDO BOOL NULL,
  PRIMARY KEY(ID_CRITERIOS_RESPONDIDOS)
);

CREATE TABLE sistemadeavaliacao.AVALIACAO (
  ID_AVALIACAO SERIAL NOT NULL,
  PROVA_ID INTEGER NOT NULL,
  PONTOS_FORTES VARCHAR NOT NULL,
  PONTOS_MELHORAR VARCHAR NOT NULL,
  PARECER VARCHAR NOT NULL,
  DATA_CONC_AVA DATE NOT NULL,
  LINK_PROVA VARCHAR NOT NULL,
  CORRIGIDA BOOL NOT NULL,
  STATUS_AVA VARCHAR(30) NOT NULL,
  PRIMARY KEY(ID_AVALIACAO)
);

CREATE TABLE sistemadeavaliacao.CANDIDATO (
  ID_CANDIDATO SERIAL NOT NULL,
  REDE_SOCIAL VARCHAR NULL,
  NOME VARCHAR(50) NOT NULL,
  EMAIL VARCHAR(50) NOT NULL,
  CONTRATADO BOOL NULL,
  TELEFONE VARCHAR NULL,
  PRIMARY KEY(ID_CANDIDATO)
);

CREATE TABLE sistemadeavaliacao.CRITERIOS (
  ID_CRITERIOS SERIAL NOT NULL,
  CRITERIOS_OBRIGATORIOS VARCHAR NOT NULL,
  DESCRICAO VARCHAR NOT NULL,
  LINGUAGEM_ID INTEGER NOT NULL,
  PRIMARY KEY(ID_CRITERIOS)
);
